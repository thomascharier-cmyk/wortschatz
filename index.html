<script>

    document.getElementById('imageInput').addEventListener('change', async (e) => {

        const file = e.target.files[0];

        if (!file) return;


        const loadingText = document.getElementById('loadingText');

        loadingText.style.display = 'block';

        

        const reader = new FileReader();

        reader.onloadend = async () => {

            const base64Image = reader.result.split(',')[1];

            const mimeType = file.type;

            

            const apiKey = document.getElementById('apiKeyInput').value.trim();

            

            if (!apiKey) {

                alert("Veuillez renseigner votre clé API Gemini pour lancer l'analyse.");

                loadingText.style.display = 'none';

                return;

            }


            try {

                const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=${apiKey}`;

                

                const promptText = "Analyses cette image de texte en allemand. Extrais absolument TOUS les mots significatifs (noms, verbes, adjectifs, mots de liaison) et traduis-les en français selon le contexte exact. Renvoie UNIQUEMENT un objet JSON avec 4 clés : 'noms', 'verbes', 'adjectifs', 'liaisons'. Structure : noms -> [{mot, traduction, base}], verbes -> [{mot, traduction, infinitif}], adjectifs -> [{mot, traduction, base}], liaisons -> [{mot, traduction, role}]. Ne mets aucun texte Markdown d'enrobage, juste le JSON pur.";


                const response = await fetch(url, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify({

                        contents: [{

                            parts: [

                                { text: promptText },

                                { inlineData: { mimeType: mimeType, data: base64Image } }

                            ]

                        }],

                        generationConfig: {

                            responseMimeType: "application/json"

                        }

                    })

                });


                if (!response.ok) throw new Error("Erreur API : " + response.statusText);

                

                const result = await response.json();

                const textResponse = result.candidates[0].content.parts[0].text;

                const data = JSON.parse(textResponse);

                

                renderTable('body-noms', data.noms, 'base');

                renderTable('body-verbes', data.verbes, 'infinitif');

                renderTable('body-adjectifs', data.adjectifs, 'base');

                renderTable('body-liaisons', data.liaisons, 'role');

                

                document.getElementById('tabsContainer').style.display = 'flex';

            } catch (error) {

                console.error(error);

                alert("Une erreur est survenue lors de l'analyse : " + error.message);

            } finally {

                loadingText.style.display = 'none';

            }

        };

        reader.readAsDataURL(file);

    });


    function renderTable(elementId, items, extraKey) {

        const tbody = document.getElementById(elementId);

        tbody.innerHTML = '';

        

        if (!items || items.length === 0) {

            tbody.innerHTML = '<tr><td colspan="3" style="text-align:center; color:#999;">Aucun élément trouvé.</td></tr>';

            return;

        }

        

        items.forEach(item => {

            const row = document.createElement('tr');

            

            const cellMot = document.createElement('td');

            cellMot.className = 'word-highlight';

            cellMot.textContent = item.mot;

            

            const cellTrad = document.createElement('td');

            cellTrad.textContent = item.traduction;

            

            const cellExtra = document.createElement('td');

            cellExtra.className = 'base-form';

            cellExtra.textContent = item[extraKey] || '-';

            

            row.appendChild(cellMot);

            row.appendChild(cellTrad);

            row.appendChild(cellExtra);

            tbody.appendChild(row);

        });

    }


    function switchTab(evt, tabId) {

        document.querySelectorAll('.tab-content').forEach(tab => tab.classList.remove('active'));

        document.querySelectorAll('.tab-btn').forEach(btn => btn.classList.remove('active'));

        document.getElementById(tabId).classList.add('active');

        evt.currentTarget.classList.add('active');

    }
